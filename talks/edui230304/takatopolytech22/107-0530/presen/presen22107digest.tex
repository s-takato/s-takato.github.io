\slidepage

\begin{layer}{120}{0}

\begin{itemize}
\item
$a$は正の定数\vspace{-1mm}
\item
任意の実数$x$について$a^x$が定まる.\\
\hspace*{1zw}$a^0=1,\ a^{-n}=\bunsuu{1}{a^n},\ a^{\frac{n}{m}}=\sqrt[m]{a^n}$\vspace{-1mm}
\item
指数法則\\
(1)\ $a^p a^{q}=a^{p+q}$\\

\end{itemize}
